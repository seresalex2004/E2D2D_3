## MSSQL – adatbázis + tábla + tesztadatok

```sql
/* ================================
   ADATBÁZIS LÉTREHOZÁSA
   ================================ */
CREATE DATABASE SutemenyBoltDB;
GO

USE SutemenyBoltDB;
GO

/* ================================
   TÁBLA LÉTREHOZÁSA
   ================================ */
CREATE TABLE Sutemenyek (
    id INT IDENTITY(1,1) PRIMARY KEY,
    nev NVARCHAR(100) NOT NULL,
    kaloria INT NOT NULL,
    ar INT NOT NULL
);
GO


INSERT INTO Sutemenyek (nev, kaloria, ar)
VALUES
  ('Csokoládétorta', 450, 1200),
  ('Almás pite', 320, 900);
GO


SELECT * FROM Sutemenyek;
```
